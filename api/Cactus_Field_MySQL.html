public function convert($type, $value)
{
	switch ($type) 
	{
		case self::BOOLEAN:
			$value = ((int) $value === 1);
		break;
		case self::INT:
		case self::YEAR:
			$value = (int) $value;
		break;
		case self::FLOAT:
			$value = (float) $value;
		break;
		case self::DATE:
			$value = \DateTime::createFromFormat("Y-m-d", $value);
		break;
		case self::TIME:
			$value = \DateTime::createFromFormat("H:i:s", $value);
		break;
		case self::DATETIME:
			$value = \DateTime::createFromFormat("Y-m-d H:i:s", $value);
		break;
		default:
	}

	return $value;
}

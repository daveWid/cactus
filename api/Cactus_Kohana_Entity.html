public function set($name, $value = null)
{
	if ( ! is_array($name))
	{
		$name = array($name => $value);
	}

	foreach ($name as $key => $value)
	{
		if ( ! $this->get($key, false) OR $this->data[$key] !== $value)
		{
			$this->modified_data[$key] = $value;
		}

		$this->data[$key] = $value;
	}

	return $this;
}

public function result()

Source Code
public function result()
{
	if ( ! $this->has_result)
	{
		$this->has_result = true;

		$this->result = $this->driver->find(array(
			$this->column => $this->value
		));
	}

	return $this->result;
}

public function driver($name = null)
{
	if ($name === null)
	{
		return $this->driver;
	}

	$this->driver = is_string($name) ? new $name : $name;
	return $this;
}
